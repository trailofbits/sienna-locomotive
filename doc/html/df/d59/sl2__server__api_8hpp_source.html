9 // An enum of response codes used by the calls below to indicate
10 // their success or failure.
11 enum class SL2Response {
12  // Nothing went wrong.
13  OK,
14  // The pipe used to talk to the server is invalid or closed.
15  BadPipe,
16  // We expected a valid path from the server, but were given something too long.
17  MaxPath,
18  // The server is not running, or not accepting connections.
19  ServerNotRunning,
20  // We expected more bytes from the server than we received.
21  ShortRead,
22  // We wrote fewer bytes to the server than we expected.
23  ShortWrite,
24  // We expected fewer bytes from the server than we received.
25  LongRead,
26  // We wrote more bytes to the server that it expected.
27  LongWrite,
28  // We tried to perform an action on the server without a run ID.
29  MissingRunID,
30  // We tried to request a run ID from the server while already having one.
31  AlreadyHasRunID,

33  MissingArenaID,
34  // We sent a request to the server, and the server failed to fulfill it.
35  ServerError,
